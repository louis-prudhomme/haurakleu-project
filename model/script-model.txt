Table User as u {
    id int [pk, increment]
    firstname varchar
    lastname varchar
    avatar_path varchar
    phone number //check form
    email varchar //check form
    password varchar //check form & hash ?
   
    is_my_user int // only 0 or 1
}

Table Performs {
    id_student int [pk, ref: < u.id]
    id_instructions int [pk, ref: < i.id]
}

Table Instructions as i {
    id int [pk]
    deadline date
}

Table Student as s {
    id int [pk, ref: - u.id]
    promotion int //check if year

    is_apprentice int // only 0 or 1
 
    id_major int [ref: < m.id]
}

Table Major as m {
    id int [pk, increment]
    label varchar
    id_major_director int [ref: < t.id]
}

Table Company as cy {
    id int [pk, increment]
    name varchar
}

Table Teacher as t {
    id int [pk, increment]
    hired date
    teaching_field varchar
}

Table Teaches {
    id_teacher int [pk, ref: < t.id]
    id_field int [pk, ref: < f.id]
}

Table Fields as f {
    id int [pk, increment]
    label varchar
}

Table Company_Tutor as ct {
    id int [pk, increment]
    job varchar
}

Table Confidentiality as c {
    id int [pk, increment]
    label varchar
}

Table Report {
    id int [pk, increment]
    title varchar
    submited date
    confidentiality_level int [ref: < c.id]

    is_pedag_vetted int // 0 or 1
    is_company_vetted int // 0 or 1
    is_final int // 0 or 1

    id_student int [ref: < s.id]
    id_pedag_tutor int [ref: < t.id]
    id_company_tutor int [ref: < ct.id]
    id_company int [ref: < cy.id]
    id_instructions int [ref: < i.id]
}
